   5// takes a function, any number of arguments, and a callback.

   9const consoleLogPrefix = '';
  10const consoleLogSuffix = '';

  12export function extractLogs (stderr:string) {
  13    const reg = new RegExp(consoleLogPrefix + ' ([\\s\\S]*?) ' + consoleLogSuffix, 'gm');
  14    const matches = stderr.match(reg); // all matches, preceeded by the entire string at matches[0]
  15    return matches? matches.slice(1).join('\n') : null;
  16}

  61            const match = err.toString().match(/(Error on line (\d+)): ([\s\S]*)/i);
  62            const errLine = !match? '*?*' : `${match[3].trim()} on line ${match[2]}: '${escapedCall[parseInt(match[2])-1]}'`; 
  63            throw {err, errLine};
  64        });
