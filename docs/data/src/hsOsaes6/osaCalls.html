  76    osaGetEmail: function(date:string) {

  77        const Mail = Application("Mail");

  78        Mail.activate();

  79        Mail.includeStandardAdditions = true;

  80        Mail.checkForNewMail();

  81

  82        const result:any[] = [];

  83        Mail.accounts().forEach((acc:any) => {

  84            const inbox = acc.mailboxes().filter((mb:any) => mb.name().toLowerCase()==="inbox")[0];

  85            const newMsgs = inbox.messages().filter((m:any) => Date.parse(m.dateReceived()) >= Date.parse(date));

  86            result.push({

  87                mailboxes:    acc.mailboxes().length,

  88                account:      acc.name(),

  89                inbox:        inbox.name(),

  90                numMsgTotal:  inbox.messages().length,

  91                numMsgNew:    newMsgs.length,

  92                dateSince:    date,

  93                msgSinceDate: newMsgs.map((m:any) => { return {from: m.sender(), subject: m.subject(), received: m.dateReceived(), id: m.id()};})

  94            });

  95        });

  96        const Term = Application("Terminal");

  97        Term.activate();

  98        return result;
