  35     * sends a message via thew OSX Messages app
  36     */
  37    osaSendMessage: function(appleIDs:string[], message:string, attachments:string[]) {    
  38        const Messages = Application("Messages");
  39        Messages.includeStandardAdditions = true;
  40        const buddies = appleIDs.map((a:string) => Messages.buddies.whose({ handle: a })[0]);
  41
  42        Messages.send(message, { to: buddies[0] });
  43        if (attachments && attachments.length>0) {
  44            attachments.forEach((a:string) => Messages.send(Path(a), { to: buddies[0] }));
  45        }
  46    },

  64            }

  72     */
  73    osaGetEmail: function(date:string) {
  74        const Mail = Application("Mail");
  75        Mail.activate();
  76        Mail.includeStandardAdditions = true;
  77        Mail.checkForNewMail();
  78
  79        const result:any[] = [];
  80        Mail.accounts().forEach((acc:any) => {
  81            const inbox = acc.mailboxes().filter((mb:any) => mb.name().toLowerCase()==="inbox")[0];
  82            const newMsgs = inbox.messages().filter((m:any) => Date.parse(m.dateReceived()) >= Date.parse(date));
  83            result.push({
  84                mailboxes:    acc.mailboxes().length,
  85                account:      acc.name(),
  86                inbox:        inbox.name(),
  87                numMsgTotal:  inbox.messages().length,
  88                numMsgNew:    newMsgs.length,
  89                dateSince:    date,
  90                msgSinceDate: newMsgs.map((m:any) => { return {from: m.sender(), subject: m.subject(), received: m.dateReceived(), id: m.id()};})
  91            });
  92        });
  93        return result;
  94    },
