  14/**
  15 * standard response handler for `osaJS` calls.
  16 * @param result 
  17 */
  18const responseHandler = (cmd:string) => {
  19    return (result:{stdout:string, logResult:string}):string => {
  20        if (result.logResult) { log.debug(`messages from executing ${cmd}: ${result.logResult}`); }
  21        if (result.stdout) { 
  22            log.debug(`call ${cmd} returned with result ${result.stdout}, ${typeof result.stdout}`); 
  23            return JSON.parse(result.stdout);
  24        } else { 
  25            log.debug(`call ${cmd} returned without stdout`); 
  26            return;
  27        }
  28    };
  29};

  31/**
  32 * standard error handler for `osaJS` calls
  33 * @param cmd 
  34 */
  35const errorHandler = (cmd:string) => {
  36    return (result: {err:string, errLine: string}):string => {
  37        log.error(result.errLine);
  38        return result.errLine;
  39    };
  40};

  49export const osaCommands = {
  50    /**
  51     * promises to send `message` to a list of buddies in `recipients` using iMessage.
  52     * @param message the message to send
  53     * @param appleIDs list of recipients in iMessage
  54     * @return a promise that resolves to the returned result 
  55     */
  56    sendMessage: (appleIDs:string[], message:string, attachments?:string[]):Promise => { 
  57        log.debug(`sending Message to ${appleIDs.join(', ')}, content '${message}', ${attachments?attachments.length:0} attachments`);
  58        return osaJS(OSXcommands.osaSendMessage, appleIDs, message, attachments)
  59            .then(responseHandler('osaSendMessage'))
  60            .catch(errorHandler('osaSendMessage ' + message));
  61    },
  62
  63    /**
  64     * sends an email to the list of addresses.
  65     * @param subject the subject of the message to send
  66     * @param addresses an array of recipient email addresses
  67     * @param content optional; the content of the message to send
  68     * @return a promise that resolves to the returned result 
  69     * @todo enable for multiple `to`s
  70     */
  71    sendEmail: (subject:string, to:string[], content='', attachments?:string[]):Promise => { 
  72        log.debug(`sending email to '${to.join(', ')}' with subject '${subject}', content '${content}', ${attachments?attachments.length:0} attachments`);
  73        return osaJS(OSXcommands.osaSendEmail, subject, to[0], content, attachments)
  74            .then(responseHandler('osaSendEmail'))
  75            .catch(errorHandler('osaSendEmail ' + subject));
  76    },
  77

 150    /**
 151     * quits an application. 
 152     * @param name the application to quit.
 153     * @return a promise that resolves to True if the application has quit, else False 
 154     */
 155    quit: (name:string):Promise => { 
 156        let i = name.lastIndexOf('/');
 157        if (i>0) { name = name.substr(i+1); }
 158        log.debug('quitting \'' + name + '\'');
 159        return osaJS(OSXcommands.osaQuit, name)
 160            .then(responseHandler('osaQuit'))
 161            .then(result => {
 162                log.debug(name + ' running: ' + result);
 163                return result? false : true;
 164            })
 165            .catch(errorHandler('osaQuit ' + name));
 166    },

 168    /**
 169     * checks if an application is running. 
 170     * @param name the application to check.
 171     * @return a promise that resolves to True if the application is running, else False 
 172     */
 173    isRunning: (name:string):Promise => {
 174        const slash = name.lastIndexOf('/');
 175        if (slash >= 0) { name = name.substring(slash+1, name.lastIndexOf('.')); }
 176        log.debug(`checking if '${name}' is running`);
 177        return cp.exec('ps -cx')
 178            .then(result => {
 179                log.debug(`isRunning result: ${result.stdout.indexOf(name)} ${result.stdout}`);
 180                return (result.stdout.indexOf(name) > 0);
 181                })
 182            .catch(errorHandler('isRunning ' + name));
 183    },
