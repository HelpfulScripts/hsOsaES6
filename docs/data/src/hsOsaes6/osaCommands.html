  15/**

  16 * standard response handler for `osaJS` calls.

  17 * @param result 

  18 */

  19const responseHandler = (cmd:string) => {

  20    return (result:{stdout:string, logResult:string}):string => {

  21        if (result.logResult) { log.debug(`messages from executing ${cmd}: ${result.logResult}`); }

  22        if (result.stdout) { 

  23            log.debug(`call ${cmd} returned with result ${result.stdout}, ${typeof result.stdout}`); 

  24            return JSON.parse(result.stdout);

  25        } else { 

  26            log.debug(`call ${cmd} returned without stdout`); 

  27            return;

  28        }

  29    };

  30};

  32/**

  33 * standard error handler for `osaJS` calls

  34 * @param cmd 

  35 */

  36const errorHandler = (cmd:string) => {

  37    return (result: {err:string, errLine: string}):string => {

  38        log.error(result.errLine);

  39        return result.errLine;

  40    };

  41};

  47/**

  48 * Container holding various OSX calls 

  49 */

  50export const osaCommands = {

  51    /**

  52     * promises to send `message` to a list of buddies in `recipients` using iMessage.

  53     * @param message the message to send

  54     * @param appleIDs list of recipients in iMessage

  55     * @return a promise that resolves to the returned result 

  56     */

  57    sendMessage: (appleIDs:string[], message:string, attachments?:string[]):Promise => { 

  58        log.debug(`sending Message to ${appleIDs.join(', ')}, content '${message}', ${attachments?attachments.length:0} attachments`);

  59        return osaJS(OSXcommands.osaSendMessage, appleIDs, message, attachments)

  60            .then(responseHandler('osaSendMessage'))

  61            .catch(errorHandler('osaSendMessage ' + message));

  62    },

  63

  64    /**

  65     * sends an email to the list of addresses.

  66     * @param subject the subject of the message to send

  67     * @param addresses an array of recipient email addresses

  68     * @param content optional; the content of the message to send

  69     * @return a promise that resolves to the returned result 

  70     * @todo enable for multiple `to`s

  71     */

  72    sendEmail: (subject:string, to:string[], content='', attachments?:string[]):Promise => { 

  73        log.debug(`sending email to '${to.join(', ')}' with subject '${subject}', content '${content}', ${attachments?attachments.length:0} attachments`);

  74        return osaJS(OSXcommands.osaSendEmail, subject, to[0], content, attachments)

  75            .then(responseHandler('osaSendEmail'))

  76            .catch(errorHandler('osaSendEmail ' + subject));

  77    },

  78

  79    /**

  80     * sends an email to the list of addresses.

  81     * @param subject the subject of the message to send

  82     * @param addresses an array of recipient email addresses

  83     * @param content optional; the content of the message to send

  84     * @return a promise that resolves to the returned result 

  85     * @todo enable for multiple `to`s

  86     */

  87    getEmail: (date:Date):Promise => { 

  88        log.debug(`getting emails`);

  89        return osaJS(OSXcommands.osaGetEmail, date)

  90            .then(responseHandler('osaGetEmail'))

  91            .catch(errorHandler('osaGetEmail'));

  92    },

  93

  94    /**

  95     * initiates a facetime call to the specified buddy.

  96     * @param appleID the buddy to initiate a facetime call to

  97     * @return a promise that resolves to the returned result 

  98     */

  99    facetime: (appleID:string) :Promise=> { 

 100        log.debug('starting facetime call with ' + appleID);

 101        return exec('open facetime://' + appleID)

 102            .catch(errorHandler('facetime ' + appleID));

 103    },

 105    /**

 106     * speaks the provided `text`.

 107     * @param text the text to speak.

 108     * @return a promise that resolves to the returned result 

 109     */

 110    say: (text:string):Promise => { 

 111        log.debug('saying \'' + text + '\'');

 112        return osaJS(OSXcommands.osaSay, text)

 113            .then(responseHandler('osaSay'))

 114            .then(() => `I said '${text}'`)

 115            .catch(errorHandler('osaSay ' + text));

 116    },

 118    /**

 119     * launches an application. 

 120     * @param name the application to launch.  

 121     * @return a promise that resolves to True if the application is running, else False 

 122     */

 123    launch: (name:string):Promise => { 

 124        log.debug('launching \'' + name + '\'');

 125        return osaJS(OSXcommands.osaLaunch, name)

 126            .then(responseHandler('osaLaunch'))

 127            .then(result => {

 128                log.debug(name + ' running: ' + result);

 129                return result? true : false;

 130            })

 131            .catch(errorHandler('osaLaunch ' + name));

 132    },

 151    /**

 152     * quits an application. 

 153     * @param name the application to quit.

 154     * @return a promise that resolves to True if the application has quit, else False 

 155     */

 156    quit: (name:string):Promise => { 

 157        let i = name.lastIndexOf('/');

 158        if (i>0) { name = name.substr(i+1); }

 159        log.debug('quitting \'' + name + '\'');

 160        return osaJS(OSXcommands.osaQuit, name)

 161            .then(responseHandler('osaQuit'))

 162            .then(result => {

 163                log.debug(name + ' running: ' + result);

 164                return result? false : true;

 165            })

 166            .catch(errorHandler('osaQuit ' + name));

 167    },

 169    /**

 170     * checks if an application is running. 

 171     * @param name the application to check.

 172     * @return a promise that resolves to True if the application is running, else False 

 173     */

 174    isRunning: (name:string):Promise => {

 175        const slash = name.lastIndexOf('/');

 176        if (slash >= 0) { name = name.substring(slash+1, name.lastIndexOf('.')); }

 177        log.debug(`checking if '${name}' is running`);

 178        return exec('ps -cx')

 179            .then(result => {

 180                log.debug(`isRunning result: ${result.stdout.indexOf(name)} ${result.stdout}`);

 181                return (result.stdout.indexOf(name) > 0);

 182                })

 183            .catch(errorHandler('isRunning ' + name));

 184    },
